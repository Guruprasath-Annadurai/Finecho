      setListening(false);

    recognition.stop();
    setListening(false);

import spacy
import dateparser
import re

nlp = spacy.load("en_core_web_sm")

def extract_data(text):
    doc = nlp(text)
    amount = None
    vendor = None
    category = "other"
    date = dateparser.parse("today")  # fallback

    # Extract amount with regex
    match = re.search(r"(\d+(\.\d+)?)\s*(pounds|euros|dollars|â‚¹)?", text)
    if match:
        amount = float(match.group(1))

    # Find vendor (proper noun near "at" or "from")
    for i, token in enumerate(doc):
        if token.text.lower() in ["at", "from"] and i+1 < len(doc):
            vendor = doc[i+1].text

    # Infer category keywords (simple rule-based for now)
    food_keywords = ["food", "groceries", "restaurant", "coffee", "Starbucks"]
    travel_keywords = ["uber", "train", "flight", "ticket", "gas"]
    text_lower = text.lower()
    if any(word in text_lower for word in food_keywords):
        category = "food"
    elif any(word in text_lower for word in travel_keywords):
        category = "travel"

    # Parse date
    for ent in doc.ents:
        if ent.label_ == "DATE":
            parsed_date = dateparser.parse(ent.text)
            if parsed_date:
                date = parsed_date
                break

    return {
        "amount": amount,
        "vendor": vendor,
        "category": category,
        "date": str(date.date())
    }

from flask import Flask, request, jsonify
from flask_cors import CORS
from nlp_parser import extract_data

app = Flask(__name__)
CORS(app)

@app.route("/parse", methods=["POST"])
def parse_text():
    data = request.get_json()
    text = data.get("transcript", "")
    if not text:
        return jsonify({"error": "No transcript provided"}), 400

    extracted = extract_data(text)
    return jsonify(extracted)

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)

import React, { useEffect, useState } from "react";
import axios from "axios";
import { Bar } from "react-chartjs-2";
import {
  Chart as ChartJS,
  CategoryScale,
  LinearScale,
  BarElement,
  Title,
  Tooltip,
  Legend
} from "chart.js";

ChartJS.register(CategoryScale, LinearScale, BarElement, Title, Tooltip, Legend);

const Dashboard = () => {
  const [data, setData] = useState([]);

  useEffect(() => {
    axios.get("http://localhost:5000/transactions").then((res) => {
      setData(res.data);
    });
  }, []);

  const categoryTotals = data.reduce((acc, txn) => {
    acc[txn.category] = (acc[txn.category] || 0) + txn.amount;
    return acc;
  }, {});

  const chartData = {
    labels: Object.keys(categoryTotals),
    datasets: [
      {
        label: "Spending by Category",
        data: Object.values(categoryTotals),
        backgroundColor: "rgba(99, 102, 241, 0.6)"
      }
    ]
  };

  return (
    <div className="p-8 bg-white min-h-screen">
      <h1 className="text-3xl font-bold mb-6 text-indigo-800">ğŸ“Š FinEcho Dashboard</h1>
      <div className="bg-gray-50 p-6 rounded shadow">
        <Bar data={chartData} />
      </div>
    </div>
  );
};

export default Dashboard;
